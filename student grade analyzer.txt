def calculate_grade(marks):
    if marks >= 90:
        return "A"
    elif marks >= 75:
        return "B"
    elif marks >= 50:
        return "C"
    else:
        return "Fail"


# Loop to process marks of 5 students
for i in range(1, 6):
    marks = int(input(f"Enter marks for student {i}: "))
    
    if 0 <= marks <= 100:
        grade = calculate_grade(marks)
        print(f"Student {i} Grade: {grade}")
    else:
        print("Invalid marks! Please enter between 0 and 100.")
